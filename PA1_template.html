# This function will return the categorical values
WeekDate <- function(date){
    t <- weekdays(date)
    if( t %in% c("Monday", "Tuesday", "Wednesday", "Thursday", "Friday"))
        return("weekday")
    return("weekend")
}

# Create a new colum with date classification (weekend or weekday)
